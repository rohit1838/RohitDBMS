DECLARE v_str VARCHAR2(20) := 'ORACLE'; v_count NUMBER := 0; v_char CHAR(1); BEGIN FOR i IN 1..LENGTH(v_str) LOOP v_char := UPPER(SUBSTR(v_str, i, 1)); IF v_char IN ('A', 'E', 'I', 'O', 'U') THEN v_count := v_count + 1; END IF; END LOOP; DBMS_OUTPUT.PUT_LINE('Number of Vowels: ' || v_count); END;

31 Write a PL/SQL block to check whether a given string is Palindrome or not. (Iterative + String Logic)
Show Answer
DECLARE v_str VARCHAR2(20) := 'MADAM'; v_reverse VARCHAR2(20); BEGIN -- Reverse the string FOR i IN REVERSE 1..LENGTH(v_str) LOOP v_reverse := v_reverse || SUBSTR(v_str, i, 1); END LOOP; -- Compare original with reversed IF v_str = v_reverse THEN DBMS_OUTPUT.PUT_LINE(v_str || ' is a Palindrome String'); ELSE DBMS_OUTPUT.PUT_LINE(v_str || ' is not a Palindrome String'); END IF; END;
